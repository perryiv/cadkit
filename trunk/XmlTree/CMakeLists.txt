
PROJECT(XmlTree)

#Include the Top Level Directory
INCLUDE_DIRECTORIES( ${CADKIT_INC_DIR} ${XERCESC_INCLUDE_DIR} "${PROJECT_SOURCE_DIR}/../" )

    Document.cpp
    Node.cpp
    Loader.cpp
    XercesLife.cpp

    Detail/RootImpl.cpp

)

	${XERCESC_LIBRARY}
)
