project (OssimLayerQt)

SET(PLUGIN_NAME "OssimLayerQt")

SET( CMakeModules "${CADKIT_INC_DIR}/CMakeModules" )
INCLUDE ( ${CMakeModules}/Cadkit.plugin.cmake)

# ------------ Find boost ----------------------
SET ( Boost_USE_FILESYSTEM TRUE )
INCLUDE( ${CMakeModules}/FindBoost.cmake)

set(COMPILE_GUARD "_COMPILING_OSSIM_LAYER_QT")
ADD_DEFINITIONS ("-D${COMPILE_GUARD}")

# Adds includes and libraries for QT.
INCLUDE(${QT_USE_FILE})
ADD_DEFINITIONS( ${QT_DEFINITIONS} )

LINK_DIRECTORIES( ${CADKIT_BIN_DIR} )

# ------------ Set Include Folders ----------------------
INCLUDE_DIRECTORIES( ${CADKIT_INC_DIR} 
                     "${PROJECT_SOURCE_DIR}/../" 
                     ${OSG_INC_DIR}
                     ${QT_INCLUDE_DIR} )

set (SOURCES
  AddOssimLayerWidget.cpp
  OssimLayerQtComponent.cpp
)

SET(MOC_HDRS
  AddOssimLayerWidget.h
)

# Make the moc files.
QT4_WRAP_CPP( MOC_SRCS ${MOC_HDRS} )

# ---------- Create a Shared Library
ADD_LIBRARY( ${PLUGIN_NAME} SHARED ${SOURCES} ${MOC_SRCS} )

CADKIT_PLUGIN()

# Link the Library
LINK_CADKIT( OssimLayerQt Usul QtTools Minerva )

TARGET_LINK_LIBRARIES( OssimLayerQt
	${QT_LIBRARIES}
	${Boost_LIBRARIES}
)
