
PROJECT(HeliosQtViewsOSG)

SET(CMakeModules "${PROJECT_SOURCE_DIR}/../../../../CMakeModules")
INCLUDE ( ${CMakeModules}/Cadkit.cmake)

# ------------ Find OpenGL Library ----------------------
FIND_PACKAGE(OpenGL)

# ------------ Set Include Folders ----------------------
#Include the Top Level Directory
INCLUDE_DIRECTORIES( ${OSG_INC_DIR} 
                     ${CADKIT_INC_DIR} 
                     "${CADKIT_INC_DIR}/Experimental"
                     "${PROJECT_SOURCE_DIR}/../../../" 
                     ${QT_INCLUDE_DIR} )

# Add definitions
ADD_DEFINITIONS ("-D_COMPILING_HELIOS_QT_VIEWS_OSG")

IF(MSVC)
	ADD_DEFINITIONS("-DNOMINMAX")
ENDIF(MSVC)

# List the Sources.
SET (SOURCES
    Viewer.cpp
    Format.cpp
    Delegate.cpp
    EditBackground.cpp
)

# These need to be moc'd.
SET(MOC_HDRS
  Viewer.h
  EditBackground.h
)

# Create the moc files.
QT4_WRAP_CPP(MOC_SRCS ${MOC_HDRS})

# Create a Shared Library
ADD_LIBRARY(HeliosQtViewsOSG SHARED ${SOURCES} ${MOC_SRCS} )

# Link the Library	
LINK_CADKIT( HeliosQtViewsOSG Usul OsgTools QtTools )

TARGET_LINK_LIBRARIES( HeliosQtViewsOSG
	${OPENGL_LIBRARIES}
   ${QT_LIBRARIES}
   ${QT_QTOPENGL_LIBRARY}
   ${OSG_LIB}
)

INSTALL(TARGETS HeliosQtViewsOSG
  RUNTIME DESTINATION bin
  LIBRARY DESTINATION bin
  ARCHIVE DESTINATION bin
)
