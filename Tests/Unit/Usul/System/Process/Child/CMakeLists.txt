
PROJECT ( UsulProcessChild )


MACRO(CADKIT_TEST_APP TARGET_NAME SOURCES CADKIT_LIBS)

	INCLUDE_DIRECTORIES( ${CADKIT_INC_DIR} )
	
	ADD_EXECUTABLE( ${TARGET_NAME} ${SOURCES} )

	# Add the target label.
	SET_TARGET_PROPERTIES( ${TARGET_NAME} PROPERTIES PROJECT_LABEL "Test: ${TARGET_NAME}" )

	# Add the debug postfix.
	SET_TARGET_PROPERTIES( ${TARGET_NAME} PROPERTIES DEBUG_POSTFIX "${CMAKE_DEBUG_POSTFIX}" )

	# Link the Library	
	LINK_CADKIT( ${TARGET_NAME} ${CADKIT_LIBS} )

ENDMACRO(CADKIT_TEST_APP TARGET_NAME SOURCES CADKIT_LIBS)

CADKIT_TEST_APP( UsulProcessChild  Main.cpp Usul )

IF(NOT WIN32)
  TARGET_LINK_LIBRARIES ( UsulProcessChild pthread )
ENDIF(NOT WIN32)
